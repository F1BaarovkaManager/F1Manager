function doPost(e) {
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getActiveSheet();
  var d = JSON.parse(e.postData.contents);
  // Hlavička (pokud je sheet prázdný)
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      'Datum', 'Hráč', 'Závod',
      'Jezdec 1','Jezdec 2','Jezdec 3',
      'Jezdec 4','Jezdec 5','Tým','Celkem'
    ]);
  }
  var drivers = d.drivers || [];
  sheet.appendRow([
    d.timestamp, d.player, d.race,
    drivers[0]||'', drivers[1]||'',
    drivers[2]||'', drivers[3]||'',
    drivers[4]||'', d.team, d.totalCost
  ]);
  return ContentService
    .createTextOutput(
      JSON.stringify({status:'ok'})
    )
    .setMimeType(
      ContentService.MimeType.JSON
    );
}
